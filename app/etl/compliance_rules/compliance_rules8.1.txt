--Rule8.1
Declare backMonth INT64 Default -15;
DECLARE StartDate Date DEFAULT DATE_ADD(DATE_TRUNC(Current_Date("Australia/Sydney"), month),  INTERVAL backMonth MONTH);
DECLARE EndDate Date DEFAULT DATE_ADD(StartDate,  INTERVAL 16 MONTH);
DECLARE RptStartDate Date DEFAULT DATE_ADD(EndDate,  INTERVAL -13 MONTH);
DECLARE MonAvgStartDate Date DEFAULT DATE("2020-10-01");
--
if Exists(Select File_Date 
          From `gcp-wow-wpay-paydathub-uat.pdh_rd_data_navigator.gfs_pdh_txn_interface`
          Where File_Date = Current_Date("Australia/Sydney") - 1
          Limit 1) Then
--Daily_Cashout
Truncate Table `gcp-wow-wpay-paydathub-uat.pdh_staging_ds.Rule8_1_Output`;
Insert into `gcp-wow-wpay-paydathub-uat.pdh_staging_ds.Rule8_1_Output`
With  Daily_PanTrn as (
           Select Upper(trim(Case 
                     When  IFNULL(rc0.Merchant_to_report,"") <> "" then rc0.Merchant_to_report
                     Else mh.division
                  End)) as Merchant,
                  trn.masked_pan as PAN,
                  Date(trn.tstamp_local) as Transactio_Date,
                  trn.cashout_total * 0.01 as Cashout
           FROM `gcp-wow-wpay-paydathub-uat.pdh_rd_data_navigator.gfs_pdh_txn_interface` trn
           Join `gcp-wow-wpay-paydathub-uat.pdh_staging_ds.ref_store_details` mh 
                on left(trn.Net_Term_ID,5) = mh.store_id 
               and mh.division is not Null 
               and mh.division <> "" 
           join gcp-wow-wpay-paydathub-uat.pdh_analytics_ds.ref_response_code r 
             on trn.act_code = r.action_code and r.approval_flag = "Y"
           left join `gcp-wow-wpay-paydathub-uat.pdh_staging_ds.Compliance_Control_Table` rc0
                   on rc0.Rule_ID = 0
                  and rc0.Attribute ="Merge" 
                  and Upper(trim(rc0.Merchant)) = Upper(trim(mh.division))
           Where trn.Net_Term_ID is not NULL 
             and trn.Net_Term_ID <> ""
             and trn.pos_crdhldr_presnt = "0"
             and trn.cashout_total <> 0
             and  Date(trn.tstamp_local) >= RptStartDate
              ),
       Daily_PanSum as (
          Select Merchant,PAN,Transactio_Date, Sum(Cashout) as Cashout
          From Daily_PanTrn
          Group By Merchant,PAN,Transactio_Date
       ),         
       merchant_Daily as (
          Select pan.Merchant,
                 pan.Transactio_Date,
                 max(Date_Diff(Date_add(DATE_TRUNC(pan.Transactio_Date,month),
                                    INTERVAL 1 MONTH),
                           DATE_TRUNC(pan.Transactio_Date,month),DAY)) Days_inMonth,
                 Sum(pan.Cashout) Cashout,
                 Sum(Case 
                        When pan.Cashout > rc3.Value
                             then 1
                        Else 0
                     End) Cashout_Exceeding_Threshold,
                 max(rc3.Value) as Threshold3
         From Daily_PanSum pan 
         left join `gcp-wow-wpay-paydathub-uat.pdh_staging_ds.Compliance_Control_Table` rc3
                  on rc3.Rule_ID = 8.1
                 and rc3.Attribute ="Single PAN Cashout Above Treshold"
         Group By pan.Merchant,
                  pan.Transactio_Date
          )
          Select md.Merchant,
                 md.Transactio_Date,
                 md.Days_inMonth,
                 md.Cashout,
                 md.Cashout_Exceeding_Threshold,
                 mco.Avg_Daily_Cashout_last_3mon,
                 rc1.value * 0.01 as Threshold1,
                 rc2.value * 0.01 as Threshold2,
                 md.Threshold3,
                 Current_Datetime() As Datetime_Created
         From merchant_Daily md   
         Left Join `gcp-wow-wpay-paydathub-uat.pdh_staging_ds.Rule8_Output` mco on mco.Merchant = md.Merchant
                                                                              and mco.FirstOfMonth = DATE_TRUNC(md.Transactio_Date,month)
         left join `gcp-wow-wpay-paydathub-uat.pdh_staging_ds.Compliance_Control_Table` rc1
                   on rc1.Rule_ID = 8
                  and rc1.Attribute ="Ratio_Cashout per month to Average Cashout last 3 months_Low"
         left join `gcp-wow-wpay-paydathub-uat.pdh_staging_ds.Compliance_Control_Table` rc2
                  on rc2.Rule_ID = 8
                 and rc2.Attribute ="Ratio_Cashout per month to Average Cashout last 3 months_High"
   Order By 2,1;
End If
